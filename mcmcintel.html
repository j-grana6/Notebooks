def EU_no_trembles_perfect_info(p1strat, p2strat, utils=utils, p1top2map=mapdict, 
                                p1tremblestrats=perfect_strat_1, p2tremblestrats=perfect_strat_2):
    """ 
    Computes the expected utility of player 1 and player 2.
    
    Nature node is hard coded to p=.5 for both outcomes.
    
    Setting trembleprobs to 0 gives standard intelligence
    """
    highukey = "H"+p1strat[0]+p2strat[mapdict[p1strat[0]]]
    lowukey =  "L"+p1strat[1]+p2strat[3 + mapdict[p1strat[1]]]
    p1eunotremb = .5 * utils[highukey][0] + .5 * utils[lowukey][0]
    p2eunotremb = .5 * utils[highukey][1] + .5 * utils[lowukey][1]
    return p1eunotremb, p2eunotremb




def intelligence_perfect_info_mc(profile, numalt=30, eudict={}):
    """
    Get's the trembling intelligence of a profile under trembling
    """
    p1strat = profile[0]
    p2strat = profile[1]
    util = EU_no_trembles_perfect_info(p1strat, p2strat)
    #print util
    p1_numalt = float(len(perfect_strat_1))
    p2_numalt = float(len(perfect_strat_2))
    num_no_betterp1 = 0
    num_no_betterp2 = 0
    for ix in np.random.choice(range(len(perfect_strat_1)), size=numalt):
        s = perfect_strat_1[ix]
        if EU_no_trembles_perfect_info(s, p2strat)[0] <= util[0]:
            num_no_betterp1 +=1
    for ix in np.random.choice(range(len(perfect_strat_2)), size=numalt):
        s= perfect_strat_2[ix]
        if EU_no_trembles_perfect_info(p1strat, s)[1] <= util[1]:
            num_no_betterp2 +=1
    return (num_no_betterp1-1)/float(numalt) * (num_no_betterp2 - 1)/float(numalt)

def mcmc_perfect_info_regular_intels(numsteps, strategyspace, power=1):
    eudict = {}
    intel_dict = {}
    expectedutilitysamps = []
    ix_prof = np.arange(len(strategyspace))
    s0 = strategyspace[0]
    s0_intel = intelligence_perfect_info_mc(s0)
    intel_dict[s0] = s0_intel
    for j in range(numsteps):

        if s1 in intel_dict.keys():
            s1_intel = intel_dict[s1]
        else:
            s1_intel = intelligence_perfect_info_mc(s1)

            s0=s1
            s0_intel = s1_intel
        else:
            expectedutilitysamps.append(EU_no_trembles_perfect_info(s0[0], s0[1]))

def EU_with_trembles_perfect_info(p1strat, p2strat, utils=utils, p1top2map=mapdict, p1trembleprob=0, 
                     p2trembleprob=0, p1tremblestrats=perfect_strat_1, p2tremblestrats=perfect_strat_2):
    """ 
    Computes the expected utility of player 1 and player 2.
    
    Nature node is hard coded to p=.5 for both outcomes.
    
    Setting trembleprobs to 0 gives standard intelligence
    """
    highukey = "H"+p1strat[0]+p2strat[mapdict[p1strat[0]]]
    lowukey =  "L"+p1strat[1]+p2strat[3 + mapdict[p1strat[1]]]
    p1eunotremb = (1-p2trembleprob) * (.5 * utils[highukey][0] + .5 * utils[lowukey][0])
    p2eunotremb = (1-p1trembleprob) * (.5 * utils[highukey][1] + .5 * utils[lowukey][1])
    if p1trembleprob + p2trembleprob ==0:
        return p1eunotremb, p2eunotremb
    ### Get the utility that is associated with p2 trembling
    p1epovern = p1trembleprob/float(len(p1tremblestrats) - 1)
    p2epovern = p2trembleprob/float(len(p2tremblestrats) - 1)
    p1tremblingeu = 0 # Utility of p1 when p2 trembles
    p2tremblingeu = 0 # Utility of p2 when p1 trembles
    for s in p2tremblestrats:
        if s != p2strat:
            highukey = "H"+p1strat[0]+s[mapdict[p1strat[0]]]
            lowukey =  "L"+p1strat[1]+s[3 + mapdict[p1strat[1]]]
            p1tremblingeu += p2epovern * (.5 * utils[highukey][0] + .5 * utils[lowukey][0])
    for s in p1tremblestrats:
        if s != p1strat:
            highukey = "H"+s[0]+p2strat[mapdict[s[0]]]
            lowukey =  "L"+s[1]+p2strat[3 + mapdict[s[1]]]
            p2tremblingeu += p1epovern * (.5 * utils[highukey][1] + .5 * utils[lowukey][1])
    return p1eunotremb + p1tremblingeu, p2eunotremb + p2tremblingeu
