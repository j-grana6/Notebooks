def EU_with_trembles_perfect_info(p1strat, p2strat, utils=utils, p1top2map=mapdict, p1trembleprob=.2, 
                     p2trembleprob=.2, p1tremblestrats=perfect_strat_1, p2tremblestrats=perfect_strat_2):
    """ 
    Computes the expected utility of player 1 and player 2.
    
    Nature node is hard coded to p=.5 for both outcomes.
    
    Setting trembleprobs to 0 gives standard intelligence
    """
    highukey = "H"+p1strat[0]+p2strat[mapdict[p1strat[0]]]
    lowukey =  "L"+p1strat[1]+p2strat[3 + mapdict[p1strat[1]]]
    p1eunotremb = (1-p2trembleprob) * (.5 * utils[highukey][0] + .5 * utils[lowukey][0])
    p2eunotremb = (1-p1trembleprob) * (.5 * utils[highukey][1] + .5 * utils[lowukey][1])
    if p1trembleprob + p2trembleprob ==0:
        return p1eunotremb, p2eunotremb
    ### Get the utility that is associated with p2 trembling
    p1epovern = p1trembleprob/float(len(p1tremblestrats) - 1)
    p2epovern = p2trembleprob/float(len(p2tremblestrats) - 1)
    p1tremblingeu = 0 # Utility of p1 when p2 trembles
    p2tremblingeu = 0 # Utility of p2 when p1 trembles
    for s in p2tremblestrats:
        if s != p2strat:
            highukey = "H"+p1strat[0]+s[mapdict[p1strat[0]]]
            lowukey =  "L"+p1strat[1]+s[3 + mapdict[p1strat[1]]]
            p1tremblingeu += p2epovern * (.5 * utils[highukey][0] + .5 * utils[lowukey][0])
    for s in p1tremblestrats:
        if s != p1strat:
            highukey = "H"+s[0]+p2strat[mapdict[s[0]]]
            lowukey =  "L"+s[1]+p2strat[3 + mapdict[s[1]]]
            p2tremblingeu += p1epovern * (.5 * utils[highukey][1] + .5 * utils[lowukey][1])
    return p1eunotremb + p1tremblingeu, p2eunotremb + p2tremblingeu

def get_trembling_intelligence_perfect_info_mc(profile, p1trembleprob=.2, p2trembleprob=.2, numalt=30):
    """
    Get's the trembling intelligence of a profile under trembling
    """
    p1strat = profile[0]
    p2strat = profile[1]
    util = EU_with_trembles_perfect_info(p1strat, p2strat, p1trembleprob=p1trembleprob, p2trembleprob=p2trembleprob)
    #print util
    p1_numalt = float(len(perfect_strat_1))
    p2_numalt = float(len(perfect_strat_2))
    num_no_betterp1 = 0
    num_no_betterp2 = 0
    for ix in np.random.choice(range(len(perfect_strat_1)), size=numalt):
        s = perfect_strat_1[ix]
        if EU_with_trembles_perfect_info(s, p2strat, p1trembleprob=p1trembleprob, p2trembleprob=p2trembleprob)[0] <= util[0]:
            num_no_betterp1 +=1
    for ix in np.random.choice(range(len(perfect_strat_2)), size=numalt):
        s= perfect_strat_2[ix]
        if EU_with_trembles_perfect_info(p1strat, s, p1trembleprob=p1trembleprob, p2trembleprob=p2trembleprob)[1] <= util[1]:
            num_no_betterp2 +=1
    return (num_no_betterp1-1)/float(numalt) * (num_no_betterp2 - 1)/float(numalt)

class TremblingIntelResultsMC(object):
    def __init__(self, res, stratprofiles = perfect_stratprofiles, perfect_strat_1=perfect_strat_1,
                 perfect_strat_2=perfect_strat_2):
        self.res = np.asarray(res)
        self.stratprofiles = stratprofiles
        self.sorted_ix = np.argsort(self.res)[::-1]
        self.perfect_strat_1 = perfect_strat_1
        self.perfect_strat_2 = perfect_strat_2
        
    def make_hist(self):
        plt.hist(self.res)
    
    def EU_perfect_info(self, p1strat, p2strat, utils=utils, p1top2map=mapdict):
        """
        No trembling EU.  USed to get Nash-eq
        """
        highukey = "H"+p1strat[0]+p2strat[mapdict[p1strat[0]]]
        lowukey =  "L"+p1strat[1]+p2strat[3 + mapdict[p1strat[1]]]
        p1eu = .5 * utils[highukey][0] + .5 * utils[lowukey][0]
        p2eu = .5 * utils[highukey][1] + .5 * utils[lowukey][1]
        return p1eu, p2eu
    
    def get_perfect_intelligence(profile):
        """
        Get's the non-trembling intelligences"""
        p1strat = profile[0]
        p2strat = profile[1]
        util = EU_perfect_info(p1strat, p2strat)
        #print util
        p1_numalt = float(len(self.perfect_strat_1))
        p2_numalt = float(len(self.perfect_strat_2))
        num_no_betterp1 = 0
        num_no_betterp2 = 0
        for s in self.perfect_strat_1:
            if EU_perfect_info(s, p2strat)[0] <= util[0]:
                num_no_betterp1 +=1
        for s in self.perfect_strat_2:
            if EU_perfect_info(p1strat, s)[1] <= util[1]:
                num_no_betterp2 +=1
        return (num_no_betterp1-1)/(p1_numalt - 1) * (num_no_betterp2 - 1)/(p2_numalt -1)
    
    def get_perfect_info_dist():
        intels_perfect = []
        for p in self.stratprofiles:
            intels_perfect.append(self.get_perfect_intelligence(p))
        return intels_perfect
    
    def eu_dist(self, power = 1, binwidth=10):
        total_prob = np.sum(self.res**power)
        probs = self.res**power/float(total_prob)
        eus = []
        for s in self.stratprofiles:
            eus.append(self.EU_perfect_info(s[0], s[1]))
        p1u = defaultdict(int)
        p2u = defaultdict(int)
        for ix, atom in enumerate(eus):
            p1u[int(atom[0]/binwidth) * binwidth ] += probs[ix]
            p2u[int(atom[1]/binwidth) * binwidth ] += probs[ix]
        return p1u, p2u
    
    def eu_hist(self, power=1, binwidth=10):
        eu_dicts = self.eu_dist(power=power, binwidth=binwidth)
        eu_subgame_perfect = [(625+250)/2. , (312. + 125.)/2.]
        eu_dicts_exact = self.eu_dist(power=power, binwidth=1)
        p1eeu = 0
        p2eeu = 0
        for k, v in eu_dicts_exact[0].iteritems():
            p1eeu += k * v
        for k, v in eu_dicts_exact[1].iteritems():
            p2eeu += k * v
        fig, [ax1, ax2] = plt.subplots(2)
        ax1.bar(eu_dicts[0].keys(), eu_dicts[0].values(), width=binwidth, color='r', alpha=.35)
        ax1.set_xlabel("Leader\'s expected utility")
        ax1.set_ylabel("Probability")
        ax1.vlines(eu_subgame_perfect[0], 0, .2)
        ax1.vlines(p1eeu, 0, .2, color='green')
        ax2.bar(eu_dicts[1].keys(), eu_dicts[1].values(), width=binwidth, color='r', alpha=.25)
        ax2.set_xlabel("Followers\'s expected utility")
        ax2.set_ylabel("Probability")
        ax2.vlines(eu_subgame_perfect[1], 0, .2)
        ax2.vlines(p2eeu, 0, .2, color='green')
        plt.tight_layout()
        fig.text(1, .5, 'The green vertical line is the expected utility of the intelligence profile. \n' + \
                        'The black line is the SPNE expected utility')
        return fig, ax1, ax2
    
    def eu_cdf(self, power=1):
        p1_dist, p2_dist = self.eu_dist( power = power)
        p1k = p1_dist.keys()
        p1v = p1_dist.values()
        p1_srtd = np.argsort(p1k)
        cumprob_p1 = np.cumsum(np.asarray(p1v)[p1_srtd])
        p2k = p2_dist.keys()
        p2v = p2_dist.values()
        p2_srtd = np.argsort(p2k)
        cumprob_p2 = np.cumsum(np.asarray(p2v)[p2_srtd])
        fig, [ax1, ax2] = plt.subplots(2)
        ax1.plot(np.asarray(p1k)[p1_srtd], cumprob_p1)
        ax1.set_xlabel("CDF of Leader\'s Utility")
        ax2.plot(np.asarray(p2k)[p2_srtd], cumprob_p2)
        ax2.set_xlabel("CDF of Followers\'s Utility")
        plt.tight_layout()
        return fig, ax1, ax2

    unvisited = list(get_off_path_info_sets(profile))
    unvisited_intels = []
    for u in unvisited:
        num_no_better = 0.
        nature = u[0]
        p1act = u[1]
        p2s_action = profile[1][(3 * int(nature=="L")) +  mapdict[p1act]]
        utility = utils[nature + p1act + p2s_action][1]

                num_no_better +=1

Justin's Idea¶

Let's try first for the game of perfect information. Basically, we define the PGT posterior as \begin{align} \frac{P(s_1, s_2)}{P(s_1', s_2')} \propto \frac{exp(\alpha EU_1(s_1, s_2))}{exp(\alpha EU_1(s_1', s_2))} \times \frac{ exp(\alpha EU_2(s_1, s_2))}{exp(\alpha EU_2(s_1, s_2'))}\times\frac{exp(\alpha EU_1(s_1, s_2'))}{exp(\alpha EU_1(s_1', s_2'))} \times \frac{exp(\alpha EU_2(s_1', s_2))}{exp(\alpha EU_2(s_1', s_2'))} \ %&\propto \frac{exp\bigg(\alpha\big(EU_1(s_1, s_2) + EU_2(s_1, s_2) +EU_1(s_1, s_2') + EU_2(s_1', s_2)\big)\bigg)}%%{exp\bigg[\alpha\bigg(EU_1(s_1', s_2) + EU_2(s_1, s_2') + EU_1(s_1', s_2')+EU_2(s_1', s_2')\big)\bigg) } \propto exp\bigg(\alpha\big(EU_1(s_1, s_2) + EU_2(s_1, s_2) +EU_1(s_1, s_2') + EU_2(s_1', s_2) - EU_1(s_1', s_2) - EU_2(s_1, s_2') - EU_1(s_1', s_2')-EU_2(s_1', s_2')\big)\bigg) \end{align}

The first two terms captures the idea of intelligence. I am more likely to play strategies that are better against a given profile. The second term is a measure of dominance. That is, strategies that are better against more strategies get more weight in the profile.

Another way to think about it is that we are doing very small intelligence computations for two profiles simultaneously. However, we are not just concerned with the ordering, we are also concerned with the magnitude (i.e. relative change in utilities).

The $\alpha$ parameter controls the responsiveness of the probability to differences in intelligences.

The following properties hold :

  1. \begin{equation} \frac{P(s_1, s_2)}{P(s_1', s_2')} = \frac{1}{\frac{P(s_1', s_2')}{P(s_1, s_2)}} \end{equation}

  2. Any strategy profile where players are both playing dominant strategies receives the highest probability.

  3. All Nash equilibrium profiles do not receive the same weight. Instead, the probability the weight they get depends on how they perform against non-equilibrium strategies.

def EU_perfect_info(p1strat, p2strat, utils=utils, p1top2map=mapdict):
    """
