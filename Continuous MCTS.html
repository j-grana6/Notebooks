Continuous Time and MCTS¶

We still have only two nodes where 'A' sends messages to 'B'

'A' sends background messages to 'B' at the rate of $\lambda_c$

The attacker has an intensity parameter, $\beta$. Intuitively, the attacker can send a message whenever he "looks at his watch" which he does at a rate of $\beta$.

The attacker does not observe the background messages

The attacker's "history" is a record of all of his message times.

We want to find a policy that maps histories to actions.

Because the history is in infinite space, we need to approximate it. Therefore, instead of remembering the exact message time, the attacker only knows it up to a finite interval of length $I$. In other words, we are gridding the observation space. (Nothing fancy)

There is a window, $w$, as well as a threshold, $k$ that determines whether or not an alarm sounds.

A malicious message is successful with probability $p$

class ContinuousTimeMCTS(object):
    def __init__(self, lambda_c=lambda_c, beta=beta, I=I, k=k, w=w, gamma=gamma, p=p, dp = dp, c= c):
        self.lambda_c = lambda_c
        self.beta = beta
        self.I = I
        self.k = k
        self.w = w
        self.gamma = gamma
        self.p =p
        self.dp = dp
        self.c =c
    def draw_s0(self):
        s0=[]
        while sum(s0) < self.w:
            s0.append(np.random.exponential(1./lambda_c))
        return list(np.cumsum(s0)[:-1])
    
    def draw_s_o_r(self, current_state, t, action):
        """
        The attacker also needs to remember all of the times that he did not send a message!
        """
        
        if action == 1:
            current_state.append(t)
            if np.random.random() < self.p:
                if len(current_state) <= k :
                    return 'Infected', t
                else:
                    return 'Alarm', t
        looking = np.random.random() < float(self.beta)/(self.beta + self.lambda_c)
        # Determines if attacker is looking or background message
        newtime = np.random.exponential(1./(self.beta + self.lambda_c))
        t+=newtime
        while not looking:
            current_state.append(t ) # Background message
            current_state = list(np.asarray(current_state)[np.asarray(current_state) >= t - self.w]) # Current state is the window
            if len(current_state) > self.k:
                return 'Alarm',t
            looking = np.random.random() < float(self.beta)/(self.beta + self.lambda_c)
            newtime = np.random.exponential(1./(self.beta + self.lambda_c))
            t += newtime
        return current_state, t
        
    def search(self, num_samps=1000):
        T= {}
        noactu = []
        actu = []
        for i in xrange(num_samps):
            s= self.draw_s0()
            r, T = self.simulate(s, [[], 0], 0.0, T, returnT=True)
            t0 =  T['[[], 0]']
            noactu.append(t0[0][0])
            actu.append(t0[1][0])
        return r, T, noactu, actu
    
    def rollout(self, s, t):
        if s == 'Alarm':
            return 0
        if s =='Infected':
            return 1
        action = int(np.random.random() < self.dp)
        new_state, newt = self.draw_s_o_r(s, t, action)
        return self.gamma * self.rollout(new_state,newt)
    
    def simulate(self, s, history, t, T = {}, returnT=False):
        """
        T is a dictionary keyed by histories. 
        values are 3 element lists.  The first is the Q
        value and the second is the count. The third keeps
        track of the total.
        
        Only terminal rewards.
        
        Might want to convert the history to asci as it should be
        less in memory...maybe?
        """
        if s == 'Alarm': 
            return 0
        if s =='Infected':
            return 1
        shist = str(history)
        if shist not in T:
            T[shist] = [[0,0], [0,0], 0] # Also stores state count
            if returnT:
                return self.rollout(s, t), T
            else:
                return self.rollout(s, t)
        this_history = T[str(history)]
        action = np.argmax([this_history[0][0] 
                            + self.c*(np.log(this_history[2]+1) / float(this_history[0][1] + 1))**.5, 
                             + this_history[1][0] 
                            + self.c*(np.log(this_history[2]+1) / float(this_history[1][1]+ 1))**.5])
        updated_history = copy.deepcopy(history)
        if action == 1:
            updated_history[0].append(int(t/self.I) * self.I)
            updated_history[1] = float(int(t/self.I) * self.I)
        else:
            #updated_history[0].append('N')
            updated_history[1] = int(t/self.I) * self.I
        inputhist = copy.deepcopy(updated_history)
        newstate, newt = self.draw_s_o_r(s, t, action)
        R = self.gamma * self.simulate(newstate, inputhist, newt, T)
        #print T
        T[str(history)][action][1]  += 1
        T[str(history)][action][0] += (R - T[str(history)][action][0])/float(T[str(history)][action][1])
        T[str(history)][2] += 1
        if returnT:
            return R, T
        else:
            return R

Notes

This framework does not fit very nicely into the SFT-Net Framework. The attacker is not setting a (finite) parameter in the generator matrix. Instead, this is more like a stochastic (extensive form) game where the player's turn is set by a random variable.

A more straightforward approach would be to have the attacker's observation be the state of all nodes and the time that he initially infects the nodes. Then for each state he chooses a rate to send malicious messages.

Multi-Node MCTS¶

In this model

  • The attacker now chooses Rates, not specific message times
  • The attacker's state is the set of infected nodes as well as the time the nodes were infected
  • The action space and the state space are infinite. To begin, we will simply grid them.

class RateMCTS(object):
    def __init__(self, lambda_c=lambda_c, action=action, k=k, w=w, gamma=gamma, p=p, dp = dp, c=c,
                 max_rate = max_rate, 
                 min_rate=min_rate, rate_step=rate_step, obs_step = obs_step, infection_states=infection_states):
        self.lambda_c = lambda_c
        self.action = action
        self.k = k
        self.w = w
        self.gamma = gamma
        self.p =p
        self.dp = dp
        self.c =c
        self.values = copy.copy(lambda_c.keys())
        action_keys = self.action.keys()
        self.values.extend(['M-B-D', 'M-A-B', 'M-A-C', 'M-C-D']) # This is a hack!
        self.max_rate = max_rate
        self.min_rate = min_rate
        self.rate_step = rate_step
        self.obs_step = obs_step
        self.infection_states = infection_states
        self.get_action_space_dict()
    
    def draw_s0(self):
        s0={}
        for edge in lambda_c.keys():
            s0_edge = []
            while sum(s0_edge) < self.w:
                s0_edge.append(np.random.exponential(1./lambda_c[edge]))
            s0[edge] = list(np.cumsum(s0_edge))[:-1]
        s0['infected'] = ['A']
        s0['infect_times'] = {'A': 0}
        s0['Infected']= False
        s0['Alarm']=False
        return s0
    

    def draw_s_o_r(self, current_state, t, action):
        """
        To draw the next state, we use the Gillespie Algorithm
        """
        if current_state['Infected'] or current_state['Alarm']:
            return 'Done', t
        new_infected = False
        rates = lambda_c.values()
        for x in self.values[4:]:
            rates.append(action[x])
        probs = np.asarray(rates)/float(np.sum(rates))
        while not new_infected:
            next_message = np.random.choice(self.values, p=probs)
            next_time = np.random.exponential(1./np.sum(rates))
            t += next_time
            current_state['N-'+ next_message[2:]].append(t)
        
            ## Here I am letting the current state being all of the  messages.
            ## However, this doesn't matter because the likelihood just uses messages
            ## in the current window.  Essentially, the state has unnecessary information.  
            ## However, at this juncture it is computationally easier to leave it as is.
            log_likelihood = 0 
            for key in current_state.keys():
                if (key != 'infected') and (key != 'infect_times') and (key != 'Alarm') and (key != 'Infected'):
                    # Be careful. there are keys infected and Infected
                    log_likelihood += np.log(poisson.pmf(np.sum(np.asarray(current_state[key]) > (t - self.w)),
                                                           self.w * self.lambda_c[key]))
            #print log_likelihood
            if log_likelihood < self.k:
                current_state['Alarm']=True
                return current_state, t
            if next_message[0] =='M':
                # If it was a malicious message
                if np.random.random() < self.p:
                    current_state['infected'].append(next_message[4])
                    current_state['infect_times'][next_message[4]] = roundres(t,self.obs_step)
                    if next_message[4] =='D':
                        current_state['Infected']=True
                    new_infected = True
        return current_state, t
        
    def search(self, num_samps=1):
        running = []
        T= {}
        for i in xrange(num_samps):
            if int(i)/500 == float(i)/500:
                print i
            s= self.draw_s0()
            r, T = self.simulate(s, [str({'A':0})], self.w, T, returnT=True)
            if i ==0:
                key = T.keys()[0]
                print len(T.keys())
            running.append(copy.copy(T[key][0]))
        return r, T, running
    
    def rollout(self, s, t):
        if s=='Done':
            return 0
        elif s['Alarm']:
            if np.random.random()<.05:
                print t, 'Alarm'
            reward= 0
            new_state='Done'
            newt=t
        elif s['Infected']:
            if np.random.random()<.05:
                print t, 'Infection'
            reward= 1
            new_state='Done'
            newt=t
        else:
            reward=0
            possible_actions = self.action_dict[tuple(sorted(s['infected']))]
            ix = np.random.randint(0, len(possible_actions))
            action = dict(zip(self.values[4:], possible_actions[ix]))
            new_state, newt = self.draw_s_o_r(s, t, action)
        return reward + self.gamma * self.rollout(new_state, newt)
    
    def simulate(self, s, history, t, T = {}, returnT=False):
        """
        T is a dictionary keyed by histories. 
        values are 3 element lists.  The first is the Q
        value and the second is the count. The third keeps
        track of the total.
        
        Only terminal rewards.
        
        Might want to convert the history to asci as it should be
        less in memory...maybe?
        """
        if s =='Done':
            return 0
        elif s['Alarm']:
            if np.random.random()<.05:
                print t, 'Alarm', 'nonew'
            reward=0
        elif s['Infected']:
            if np.random.random()<.05:
                print t, 'Infection', 'nonew'
            reward=1
        else:
            reward=0
        shist = str(history)
        
        if (shist not in T):
            #print s['infected']
            T[shist] = [[0]*len(self.action_dict[tuple(sorted(s['infected']))]),
                        [0]*len(self.action_dict[tuple(sorted(s['infected']))]), 0] #Stores Utilities and counts
            if returnT:
                return self.rollout(s, t), T
            else:
                return self.rollout(s, t)
        
        this_history = T[str(history)]
        utils = np.asarray(this_history[0])
        counts = np.asarray(this_history[1])
        total = np.asarray(this_history[2])
        action_ix = np.argmax((np.asarray(utils) + self.c*(np.log(total+1)/(counts+1)))**.5)
        new_action = copy.deepcopy(dict(zip(self.values[4:],self.action_dict[tuple(sorted(s['infected']))][action_ix])))
        updated_history = copy.deepcopy(history)
        updated_history.append(new_action)
        #print s['infect_times'], new_action
        if not (s['Alarm'] or s['Infected']):
            updated_history.append(copy.deepcopy(s['infect_times']))
        newstate, newt = self.draw_s_o_r(s, t, new_action)
        inputhist = copy.deepcopy(updated_history)
        R = reward + self.gamma * self.simulate(newstate, inputhist, newt, T)
        T[str(history)][1][action_ix]  += 1
        T[str(history)][0][action_ix] += (R - T[str(history)][0][action_ix])/float(T[str(history)][1][action_ix])
        T[str(history)][2] += 1
        if returnT:
            return R, T
        else:
            return R

    def get_action_space_dict(self):
        action_dict = {}
        for infected_nodes in self.infection_states:
            possible_actions = []
            for key in self.values[4:]:
                if key[-1] in infected_nodes:
                    possible_actions.append([0])
                elif key[-3] not in infected_nodes:
                    possible_actions.append([0])
                else:
                    possible_actions.append(list(np.arange(self.min_rate, self.max_rate, self.rate_step)))
            action_list = list(itertools.product(*possible_actions))
            try:
                action_list.remove((0,0,0,0))
            except Exception:
                pass
            action_dict[tuple(sorted(infected_nodes))] = action_list
        self.action_dict = action_dict
        
def roundres(num, res):
    return round(num / res) * res
