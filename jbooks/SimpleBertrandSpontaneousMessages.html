Event Driven Pricing Game¶

The Market¶

  • The demand that firm $i$ faces is given by $Q_i = a_s-\beta p_i +\gamma p_j$, where $a$ is determined by the state of the market.
  • The market spontaneously changes state at a rate of $\lambda_m$
  • The market informs each player of the state by sending messages to player 1 and player 2 at a rate of $\mu_1$ and $\mu_2$, respectively.

Players¶

  • 2 human controlled SFTs that represent firms
  • The players only have one state
  • The players choose a price upon receiving a message from the market by sending messages to the market. Prices go to market at a rate of $\nu_1$ and $\nu_2$.
  • In short, the player's can receive messages from the market at different times and their messages (which are prices) can reach the market at different times.

Equilibrium¶

  • The quantity is determined by the state of the market and the most recent message from both players.

  • Rewards are accumulated as profit, computed as $P\times Q-CQ$, per unit time. $C$ is an exogenously specified per-unit cost.

Other Points¶

  • The game is infinite horizon
  • Rewards are discounted

function receive_message_and_react(sft::Player, message::Message)
    #     This function will have a player receive a message and then emit
    #     a message and possibly change its state.  If the current state of the players
    #     and the received message has not yet occured for the player, the player will 
    #     pick a message uniformly random.  If the player has received the message in 
    #     his current state, he will do UCT.  
        
    #     Note, this function does NOT update the MCTS nodes, it only picks the player's
    #     action.

    #     Output is a message.  The Player object may also change its currentstate method
    if ~ sft.trembling
        action = sft.strategy[message, sft.currentstate]
        sft.currentstate = action[2]
    else
        if rand() < sft.trembleprob
            action = (sample(sft.messages), sample(sft.statespace))
            sft.currentstate = action[2]
        else
            action = sft.strategy[message, sft.currentstate]
            sft.currentstate = action[2]
        end
    end
    return action
end;

function generate_random_strategy(messages_to_send::Array{Message, 1}, messages_to_receive::Array{Message, 1},

    strategy = Dict{(Message, Any), (Message, Any)}()
    for instate in statespace
        for m_in in messages_to_receive
            messageout = sample(messages_to_send)
            newstate = sample(statespace)
            strategy[(m_in, instate)] = (messageout, newstate)
        end
    end

A note on profits¶

We assume that profits are accumulated continuously. Therefore, total utility is given by \begin{equation} \int_0^\infty e^{-\beta t}\pi(t)dt. \end{equation}

However, profits only change when either the market changes state (demand shock) or a new product quantity is received by the market.
Therefore, if we let $\tau_1, \tau_2...$ denote the times that the profit stream changes. Total utility is computed as

\begin{equation} \sum_{i=1}^{\infty}\int_{\tau_i}^{\tau_{i+1}} e^{-\beta t}\pi(t)dt. \end{equation}

where $\pi(t)$ is constant in any interval $(\tau_i, \tau_{i+1})$. This implies that for any interval, the contribution to total utility is given by

\begin{equation} \frac{e^{-\tau_i \beta} - e^{-\tau_{i+1} \beta}}{\beta}\pi \end{equation}

function get_profits(net::SFTNet, tau1::Float64, tau2::Float64)
    quantities = get_quantities(net.market)
    p1_profit = quantities[1] * net.market.pricesinmarket["Player 1"] - 
        net.sftdict["Player 1"].cost * quantities[1]
    p2_profit = quantities[2] * net.market.pricesinmarket["Player 2"] - 
    net.sftdict["Player 2"].cost * quantities[2]
    p1_discounted_profit = ( (e^(-tau1 * net.sftdict["Player 1"].discount) - e^(-tau2 * net.sftdict["Player 1"].discount)) * 
            p1_profit) /net.sftdict["Player 1"].discount
    p2_discounted_profit = ( (e^(-tau1 * net.sftdict["Player 2"].discount) - e^(-tau2 * net.sftdict["Player 2"].discount)) * 
            p2_profit) /net.sftdict["Player 2"].discount
    return [p1_discounted_profit, p2_discounted_profit]
end;

function simulate(net::SFTNet, t_end::Int64)
    t = 0.
    total_profits = [0,0]
    possible_events = get_possible_events(net)
    while t < t_end
        probs = WeightVec(Float64[e.rate for e in possible_events])
        event = sample(possible_events, probs)
        #print(event, "\n \n")
        # An event is either a spontaneous transm/transi or a message received
        if (typeof(event)==Message)
            ix = findfirst(net.messagesintransit, event)
            splice!(net.messagesintransit, ix) # Removes the messge from the network
            # If the event is the player receiving a message
            if event.sender == "Market"
                player = net.sftdict[event.receiver]
                # "player" is now the Player type        
                action = handle_event(event, net)
                new_messages = action[1]
                net.messagesintransit = [net.messagesintransit, new_messages]
            else
                # If the event is the market receiving a message
                handle_event(event, net) # The market changes quantities in market
            end
        else
            new_messages = handle_event(event, net)
            net.messagesintransit = [net.messagesintransit, new_messages]
        end
        possible_events = get_possible_events(net)
        timeincrement = get_time_increment(possible_events)
        immediate_profits = get_profits(net, t, t+timeincrement)
        t += timeincrement
        total_profits += immediate_profits
    end
    return total_profits
end;

Confidence Interval PGT¶

1) Sample a joint strategy profile and compute the expected utility "almost exactly" (i.e. with a large number of runs).

2) For each alternative joint strategy profile, begin sampling the stochastic process to compute the expected utility.

3) (Suppose we are computing the intelligence for player 1's profile). After each sample of the network, compute the confidence interval of player 1's expected utility under his alternative strategy at a pre-specified significance level.

4) If the expected utility of player 1's "outerloop" strategy is outside of the confidence interval of player 1's expected utility under the alternative strategy, classify the alternative strategy as either better or worse than the outerloop strategy according to whether the outerloop strategy is above or below the confidence interval.

5) Set a predefined maximum number of simulations in case the outerloop expected utility is never outside of the confidence interval.

function is_worse(net::SFTNet, t_max::Int64, nruns::Int64, baseeu::Float64, player::Int64,
    confintmin=50, z=2.576)
    ## The player is the index to compare in the simulate output.
    runningeu = 0
    sse = 0
    s = 1
    running_sum_squares= 0
    while s < nruns
        utility = simulate(net, t_max)[player]
        e = utility - runningeu
        runningeu += e/s
        sse += e* (utility - runningeu)
        variance = sse/s
        s += 1
        if s >confintmin
            # confintmin large enough so that we can approximate the mean by a normal distribution, not t.
            conf_l = runningeu - z*(variance/s)^.5
            conf_u = runningeu + z*(variance/s)^.5
            if (baseeu < conf_l) | (baseeu>conf_u)
                s = nruns #Exit Loop
            end
        end
    end
    return runningeu
end;

            net.sftdict["Player 1"].trembling=false
            net.sftdict["Player 2"].trembling=true
            p1altstrat = generate_random_strategy(p1possible, messages_to_p1, 
                p1statespace)
            net.sftdict["Player 1"].strategy = p1altstrat
            eu_inner = is_worse(net, eu_tmax, eu_nsims, eu_outer[1], 1)
            if eu_inner < eu_outer[1]
                num_no_betterp1 += 1
            end
        end
        net.sftdict["Player 1"].strategy = p1_strategy
        for a in 1:numalt
            net.sftdict["Player 1"].trembling=true
            net.sftdict["Player 2"].trembling=false
            p2altstrat = generate_random_strategy(p2possible, messages_to_p2, 
                p2statespace)
            net.sftdict["Player 2"].strategy = p2altstrat
            eu_inner = is_worse(net, eu_tmax, eu_nsims, eu_outer[2], 2)
            if eu_inner < eu_outer[2]
                num_no_betterp2 += 1
            end
        end
        intel = (num_no_betterp1/numalt) * (num_no_betterp2/numalt)
        push!(intels, intel)
    end
    return intels, p1_utilities, p2_utilities, sampled_strats
end;

function plot_intelligence_cdf(intels::Array, utility::Array, exponent=1.)
    intels=intels.^exponent
    normedprobs = intels/sum(intels)
    d=Dict{Float64, Float64}()
    # DefaultDict is not working right now.  I will just do
    # this the poor man's way
    for util_ix in 1:length(utility)
        d[utility[util_ix]] = 0
    end
    for util_ix in 1:length(utility)
        d[utility[util_ix]] += normedprobs[util_ix]
    end
    u2 = Float64[]
    p2 = Float64[]
    for (key, val) in d
        push!(u2, key)
        push!(p2, val)
    end
    std_util = sort(u2)
    std_util_ix = sortperm(u2)
    std_prob = [p2[ix] for ix in std_util_ix]
    return std_util, cumsum(std_prob)
end

Though the trembling hand PGT approach does have its limitations, let's see what happens when we allow the players to tremble.


In this case, a tremble is not a complete strategy tremble but a tremble at random action times.  More specifically, when computing the intellgience of a strategy profile and computing the expected utility for player 1, player 2 plays an action according to his strategy with probability $p$.  Then, with probability $(1-p)$ he selects (uniformly) one of his other actions.
