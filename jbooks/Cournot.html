Event Driven Production Game¶

The Market¶

  • The state of the market is product demand of the form $P = a_s-\beta_s Q$
  • The market spontaneously changes state at a rate of $\lambda_m$
  • The market sends messages to the player 1 and 2 at a rate of $\mu_1$ and $\mu_2$, respectively.

Players¶

  • 2 human controlled SFTs
  • The players only have one state
  • The player's choose production quantities upon receiving a message from the market by sending messages to the market. Products go to market at a rate of $\nu_1$ and $\nu_2$

Equilibrium¶

  • The price is determined by the state of the market and the most recent message from both players.

  • Rewards are accumulated as profit, computed as $P\times Q-CQ$, per unit time. $C$ is an exogenously specified per-unit cost.

Other Points¶

  • The game is infinite horizon
  • Rewards are discounted

function receive_message_and_react(sft::Player, message::Message)
    #     This function will have a player receive a message and then emit
    #     a message and possibly change its state.  If the current state of the players
    #     and the received message has not yet occured for the player, the player will 
    #     pick a message uniformly random.  If the player has received the message in 
    #     his current state, he will do UCT.  
        
    #     Note, this function does NOT update the MCTS nodes, it only picks the player's
    #     action.

    #     Output is a message.  The Player object may also change its currentstate method 
    visited = haskey(sft.MCTSvalues, (message, sft.currentstate))
    if  ~ visited
        sft.MCTSvalues[(message, sft.currentstate)] = zeros(length(sft.actions))
        sft.MCTScounts[(message, sft.currentstate)] = zeros(length(sft.actions))
        action = sample(sft.actions)
        sft.currentstate = action[2]
        return action
    else
        counts = sft.MCTScounts[(message, sft.currentstate)] + 1 # To determine UCT factor
        uct_factor = sft.c * (log (sum(counts) ) ./ counts ).^.5 
        ix = indmax(sft.MCTSvalues[(message, sft.currentstate)] + uct_factor)
        action = sft.actions[ix]
        sft.currentstate = action[2]
        return action
    end
end;

function build_net_for_experiment(mkt_to_p1::Float64, mkt_to_p2::Float64, market_state_space, p1quantities,
    p2quantities, p1tomarket::Float64, p2tomarket::Float64, market_change_rate::Float64, p1c::Float64,
    p2c::Float64, discount::Float64 )
    # DEFINE AND COLLECT ALL POSSIBLE MESSAGES IN THE GAME#
    p1messages = Message[]
    p2messages = Message[]
    marketmessages = Message[]
    # First let's generate market messages to each player
    for s in market_state_space
        push!(marketmessages, Message("Market", "Player 1", s, mkt_to_p1))
        push!(marketmessages, Message("Market", "Player 2", s, mkt_to_p2))
    end

    for q in p1quantities
        push!(p1messages, Message("Player 1", "Market", q, p1tomarket))
    end

    for q in p2quantities
        push!(p2messages, Message("Player 2", "Market", q, p2tomarket))
    end    
    # Now generate message from players to market
    # END DEFINE ALL MESSAGES#

    ## DEFINE AND COLLECT ALL SPONTANEOUS EVENTS FOR THE MARKET
    market_spon_dict = Dict()
    for ix in 1:length(market_state_space)
        if ix ==1
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate, market_state_space[2], marketmessages[3:4])
            push!(market_spon_dict[market_state_space[ix]], s_event)

        elseif ix == length(market_state_space)
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate, market_state_space[(length(market_state_space) -1)],
                [marketmessages[(length(marketmessages) - 3) : (length(marketmessages) - 2)]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
        else
            start_ix = 2*ix +1
            end_ix = 2*ix+2
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate/2., market_state_space[(ix+1)],
                [marketmessages[start_ix : end_ix]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
            start_ix = 2*ix-3
            end_ix = 2*ix-2
            s_event = Spont("Market", market_change_rate/2., market_state_space[(ix+-1)],
                [marketmessages[start_ix : end_ix]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
        end
    end
    ## END MARKET SPONTANEOUS EVENTS

    #CREATE SFTs#
    p1 = Player(discount, p1messages, p1c, 1., ["Only State"], "Player 1");
    p2 = Player(discount, p2messages, p2c, 1., ["Only State"], "Player 2");
    market = Market(market_change_rate, marketmessages, Dict{}(), market_state_space, (50,1), "Market", market_spon_dict)
    #END CREATE SFTs#
    sft_dict = {"Player 1"=> p1, "Player 2"=>p2, "Market"=>market}

    # CREATE AND INTIALIZE THE SFTNET#
    net = SFTNet([p1, p2], market, Message[], sft_dict)
    # END CREATE AND INITIALIZE THE SFTNET#;
    return net
end

function MCTS(net::SFTNet, num_samps::Int64, depth::Int64, burnout::Int64)
    for s in 1:num_samps
        resetnet(net)
        t = 0
        d = 0
        possible_events = get_possible_events(net)
        simulate(net, 0 ,depth, possible_events, burnout)
    end
end;

function simulate(net::SFTNet, d::Int64, depth::Int64, possible_events::Array, burnout::Int64)
    if d > depth
        return [0., 0.]
    end
    probs = WeightVec(Float64[e.rate for e in possible_events])
    event = sample(possible_events, probs)
    #print(event, "\n")
    if (typeof(event)==Message)
        ix = findfirst(net.messagesintransit, event)
        splice!(net.messagesintransit, ix)
        # Let's just handle the cases separately
        # and not do anything fancy.  First we
        # handle the case where there is a player acting
        if event.sender == "Market" # CHANGE TO event.receiver != Market (indicating player node)
            player = net.sftdict[event.receiver]
            #print(player.MCTSvalues, "\n")
            #print(player.MCTScounts, "\n")
            #print("DEPTH IS", d, "\n")
            currentstate = copy(player.currentstate)
            notfirst = haskey(player.MCTSvalues, (event, player.currentstate))
            #print(notfirst)
            action = handle_event(event, net)
            new_messages = action[1]
            net.messagesintransit = [net.messagesintransit, new_messages]
            action_ix = findfirst(player.actions, action)
            ## At this point, we know how the player acted
            ## now we need to compute the reward from the time of
            ## the action until the next event
            ## This requires determining the time until the next event
            next_possible_events = get_possible_events(net)
            timeincrement = get_time_increment(next_possible_events)
            #print(timeincrement, "\n")
            immediate_profits = get_profits(net, timeincrement)
            #print(immediate_profits, "\n")
            if (depth - d >burnout) 
                # We omit last visits so we do not downward bias results
                player.MCTScounts[(event, currentstate)][action_ix] += 1
                # Put this above so UCT is accurate while traversing
                # within a run...probably doesn't make a difference
            end
            total_profits = immediate_profits + simulate(net, d+1, depth, next_possible_events, burnout) ./
                [(1 + net.sftdict["Player 1"].discount)^timeincrement, 
                (1 + net.sftdict["Player 2"].discount)^timeincrement]
            #print("THE TOTAL PROFITS ARE   ", total_profits, "\n")
            if player.name == "Player 1"
                profit = total_profits[1]
            else
                profit = total_profits[2]
            end
            ## Now we can update the MCTS Value for the player that took an action
            if (depth - d >burnout)
                # We omit last visits so we do not downward bias results
                player.MCTSvalues[(event, currentstate)][action_ix] +=  
                     (profit - player.MCTSvalues[(event, currentstate)][action_ix])/ 
                    player.MCTScounts[(event, currentstate)][action_ix]
            end
            #print(total_profits, "\n")
            return total_profits
        else
            new_messages = handle_event(event, net)
            net.messagesintransit = [net.messagesintransit, new_messages]
            next_possible_events = get_possible_events(net)
            timeincrement = get_time_increment(next_possible_events)
            #print(timeincrement, "\n")
            immediate_profits = get_profits(net, timeincrement)
            #print(immediate_profits, "\n")
            total_profits = immediate_profits + simulate(net, d+1, depth, next_possible_events, burnout) ./
                [(1 + net.sftdict["Player 1"].discount)^timeincrement, 
                (1 + net.sftdict["Player 2"].discount)^timeincrement]
            #print(total_profits, "\n")
            return total_profits
        end
    else
        new_messages = handle_event(event, net)
        net.messagesintransit = [net.messagesintransit, new_messages]
        next_possible_events = get_possible_events(net)
        timeincrement = get_time_increment(next_possible_events)
        #print(timeincrement, "\n")
        immediate_profits = get_profits(net, timeincrement)
        #print(immediate_profits, "\n")
        total_profits = immediate_profits + simulate(net, d+1, depth, next_possible_events, burnout) ./
                [(1 + net.sftdict["Player 1"].discount)^timeincrement, 
                (1 + net.sftdict["Player 2"].discount)^timeincrement]
        #print(total_profits, "\n")
        return total_profits
    end
end;

function handle_event(event::Any, net::SFTNet)
    if typeof(event) == Spont
        return receive_message_and_react(net.market, event)
    elseif event.receiver=="Market"
        return receive_message_and_react(net.market, event)
    else
        return receive_message_and_react(net.sftdict[event.receiver], event)
    end
end;

function get_profits(net::SFTNet, duration::Float64)
    price = get_price(net.market)
    p1_profit = price * net.market.quantitiesinmarket["Player 1"] - 
        net.sftdict["Player 1"].cost * net.market.quantitiesinmarket["Player 1"]
    p2_profit = price * net.market.quantitiesinmarket["Player 2"] - 
        net.sftdict["Player 2"].cost * net.market.quantitiesinmarket["Player 2"]
    p1_discounted_profit = p1_profit /net.sftdict["Player 1"].discount * 
        (1-e^(-net.sftdict["Player 1"].discount * duration))
    p2_discounted_profit = p2_profit /net.sftdict["Player 2"].discount * 
        (1-e^(-net.sftdict["Player 2"].discount * duration))
    return [p1_discounted_profit, p2_discounted_profit]
end;

function simulate_network(net, n, depth)
    # This function will simulate
    # the network n times according to the action
    # that is best in MCTSvalues.
    # It will return a list of expected
    # utilities for each player.
    t = 0
    for p in net.players
        p.c = 0 #Change c so that optimal action is picked
    end
    eus = Array[]
    for samp in 1:n
        times = Float64[]
        profits_p1 = Float64[]
        profits_p2 = Float64[]
        t = 0 
        d = 0 
        resetnet(net)
        while d < depth
            push!(times, t)
            possible_events = get_possible_events(net)
            timeincrement = get_time_increment(possible_events)
            profit_since_last_event = get_profits(net, timeincrement)
            push!(profits_p1, profit_since_last_event[1])
            push!(profits_p2, profit_since_last_event[2])
            probs = WeightVec(Float64[e.rate for e in possible_events])
            event = sample(possible_events, probs)
            if typeof(event)==Message
                ix = indexin([event], net.messagesintransit)[1]
                splice!(net.messagesintransit, ix)
            end
            if typeof(event)==Message
                if event.receiver != "Market"
                    new_messages = handle_event(event, net)[1]
                else
                    new_messages = handle_event(event, net)
                end
            else
                new_messages = handle_event(event,net)
            end
            net.messagesintransit = [net.messagesintransit, new_messages]
            d += 1
            t += timeincrement
        end
        push!(eus, [sum(profits_p1.*e.^ ( - net.sftdict["Player 1"].discount.* times)), 
                    sum(profits_p2.*e.^ ( - net.sftdict["Player 2"].discount.* times))])
    end
    return eus
end;
            
function makehist(net::SFTNet, n::Int64, depth::Int64)
    res = simulate_network(net, n, depth)
    PyPlot.plt.hist([x[1] for x in res], bins=[range(1,25, 60)], alpha=.8)
    PyPlot.plt.hist([x[2] for x in res], bins=[range(1,25, 60)], alpha=.4);
    return res
end;

function print_strats(net::SFTNet)
    p1_input = Any[]
    p1_action = Any[]
    print("PLAYER 1's STRATEGY", "\n")
    for k in keys(net.players[1].MCTSvalues)
        ix = indmax(net.players[1].MCTSvalues[k])
        push!(p1_input,  k[1].content[2])
        push!(p1_action, net.players[1].actions[ix][1].content)
    end
    p1ix = sortperm(p1_input)
    for ix in p1ix
        print(p1_input[ix], " => ", p1_action[ix], "\n")
    end
    
    p2_input = Any[]
    p2_action = Any[]
    print("PLAYER 2's STRATEGY", "\n")
    for k in keys(net.players[2].MCTSvalues)
        ix = indmax(net.players[2].MCTSvalues[k])
        push!(p2_input, k[1].content)
        push!(p2_action, net.players[2].actions[ix][1].content)
    end
    p2ix = sortperm(p2_input)
    for ix in p2ix
        print(p2_input[ix], " => ", p2_action[ix], "\n")
    end
end;

function build_net_for_experiment_2(mkt_to_p1::Float64, market_state_space, 
                                    p1quantities, p2quantities, p1tomarket::Float64, 
                                    p2tomarket::Float64,
                                    p1top2::Float64,  market_change_rate::Float64, p1c::Float64,
                                    p2c::Float64, discount::Float64 )
    
    # DEFINE AND COLLECT ALL POSSIBLE MESSAGES IN THE GAME#
    p1messages = Any[]
    p2messages = Any[]
    marketmessages = Any[]
    # First let's generate market messages to each player
    for s in market_state_space
        push!(marketmessages, Message("Market", "Player 1", s, mkt_to_p1))
    end

    for q in p1quantities
        mset = Message[]
        push!(mset, Message("Player 1", "Market", q, p1tomarket))
        push!(mset, Message("Player 1", "Player 2", q, p1top2))
        push!(p1messages, mset)
    end

    for q in p2quantities
        push!(p2messages, Message("Player 2", "Market", q, p2tomarket))
    end    
    # Now generate message from players to market
    # END DEFINE ALL MESSAGES#

    ## DEFINE AND COLLECT ALL SPONTANEOUS EVENTS FOR THE MARKET
    market_spon_dict = Dict()
    for ix in 1:length(market_state_space)
        if ix ==1
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate, market_state_space[2], [marketmessages[2]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
        elseif ix == length(market_state_space)
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate, market_state_space[(ix -1)],
            [marketmessages[(ix - 1 )]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
        else
            market_spon_dict[market_state_space[ix]] = Spont[]
            s_event = Spont("Market", market_change_rate/2., market_state_space[(ix+1)],
                [marketmessages[(ix + 1)]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
            s_event = Spont("Market", market_change_rate/2., market_state_space[(ix-1)],
                [marketmessages[(ix - 1)]])
            push!(market_spon_dict[market_state_space[ix]], s_event)
        end
    end
    ## END MARKET SPONTANEOUS EVENTS
    #CREATE SFTs#
    p1 = Player(discount, p1messages, p1c, 1., ["Only State"], "Player 1");
    p2 = Player(discount, p2messages, p2c, 1., ["Only State"], "Player 2");
    market = Market(market_change_rate, marketmessages, Dict{}(), market_state_space, (50,1), "Market", market_spon_dict)
    #END CREATE SFTs#
    sft_dict = {"Player 1"=> p1, "Player 2"=>p2, "Market"=>market}

    # CREATE AND INTIALIZE THE SFTNET#
    net = SFTNet([p1, p2], market, Message[], sft_dict)
    # END CREATE AND INITIALIZE THE SFTNET#;
    return net
end
